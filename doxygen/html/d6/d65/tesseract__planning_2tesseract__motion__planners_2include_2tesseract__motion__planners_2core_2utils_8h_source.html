50 inline tesseract_common::VectorIsometry3d interpolate(const Eigen::Isometry3d& start,
51  const Eigen::Isometry3d& stop,
52  int steps)
53 {
54  // Required position change
55  Eigen::Vector3d delta_translation = (stop.translation() - start.translation());
56  Eigen::Vector3d start_pos = start.translation();
57  Eigen::Affine3d stop_prime = start.inverse() * stop;
58  Eigen::AngleAxisd delta_rotation(stop_prime.rotation());
59 
60  // Step size
61  Eigen::Vector3d step = delta_translation / steps;
62 
63  // Orientation interpolation
64  Eigen::Quaterniond start_q(start.rotation());
65  Eigen::Quaterniond stop_q(stop.rotation());
66  double slerp_ratio = 1.0 / steps;
67 
68  tesseract_common::VectorIsometry3d result;
69  Eigen::Vector3d trans;
70  Eigen::Quaterniond q;
71  Eigen::Isometry3d pose;
72  result.reserve(static_cast<size_t>(steps) + 1);
73  for (unsigned i = 0; i <= static_cast<unsigned>(steps); ++i)
74  {
75  trans = start_pos + step * i;
76  q = start_q.slerp(slerp_ratio * i, stop_q);
77  pose = (Eigen::Translation3d(trans) * q);
78  result.push_back(pose);
79  }
80  return result;
81 }
