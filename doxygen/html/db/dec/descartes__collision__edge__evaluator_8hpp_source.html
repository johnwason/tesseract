78 template <typename FloatType>
79 bool DescartesCollisionEdgeEvaluator<FloatType>::evaluate(
80  const descartes_light::Rung_<FloatType>& from,
81  const descartes_light::Rung_<FloatType>& to,
82  std::vector<typename descartes_light::LadderGraph<FloatType>::EdgeList>& edges)
83 {
84  assert(from.data.size() % dof_ == 0);
85  assert(to.data.size() % dof_ == 0);
86  const auto n_start = from.data.size() / dof_;
87  const auto n_end = to.data.size() / dof_;
88 
89  // Allocate
90  edges.resize(n_start);
91 
92  for (std::size_t i = 0; i < n_start; ++i)
93  {
94  const auto* start_vertex = from.data.data() + dof_ * i;
95  for (std::size_t j = 0; j < n_end; ++j)
96  {
97  const auto* end_vertex = to.data.data() + dof_ * j;
98 
99  // Consider the edge:
100  considerEdge(edges[i], start_vertex, end_vertex, j, allow_collision_);
101  }
102  }
103 
104  for (const auto& rung : edges)
105  if (!rung.empty())
106  return true;
107 
108  return false;
109 }

111 template <typename FloatType>
112 void DescartesCollisionEdgeEvaluator<FloatType>::considerEdge(
113  typename descartes_light::LadderGraph<FloatType>::EdgeList& out,
114  const FloatType* start,
115  const FloatType* end,
116  std::size_t next_idx,
117  bool find_best)
118 {
119  // Happens in two phases:
120  // 1. Compute the transform of all objects
121  tesseract_common::TrajArray segment(2, dof_);
122  for (size_t i = 0; i < dof_; ++i)
123  {
124  segment(0, static_cast<long>(i)) = start[i];
125  segment(1, static_cast<long>(i)) = end[i];
126  }
127 
128  std::vector<tesseract_collision::ContactResultMap> discrete_results;
129  std::vector<tesseract_collision::ContactResultMap> continuous_results;
130  bool discrete_in_contact = discreteCollisionCheck(discrete_results, segment, find_best);
131  bool continuous_in_contact = continuousCollisionCheck(continuous_results, segment, find_best);
132 
133  if (!discrete_in_contact && !continuous_in_contact)
134  out.emplace_back(0, next_idx);
135  else if (!discrete_in_contact && continuous_in_contact && find_best)
136  out.emplace_back(collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance, next_idx);
137  else if (discrete_in_contact && !continuous_in_contact && find_best)
138  out.emplace_back(collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance, next_idx);
139  else if (discrete_in_contact && continuous_in_contact && find_best)
140  {
141  double d = collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance;
142  double c = collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance;
143  out.emplace_back(std::max(d, c), next_idx);
144  }
145 }
