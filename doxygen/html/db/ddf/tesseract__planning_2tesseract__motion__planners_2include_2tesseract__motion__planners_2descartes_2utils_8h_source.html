71  const FloatType z_axes_sample_resolution = 60 * M_PI / 180)
72 {
73  UNUSED(x_sample_resolution);
74  UNUSED(y_sample_resolution);
75  UNUSED(z_sample_resolution);
76  tesseract_motion_planners::PoseSamplerFn tool_pose_sampler = nullptr;
77  if (wp->getType() == WaypointType::CARTESIAN_WAYPOINT)
78  {
79  CartesianWaypoint::ConstPtr cwp = std::static_pointer_cast<const CartesianWaypoint>(wp);
80  if (wp->getCoefficients().size() == 0 || (wp->getCoefficients().array() > 0).all()) // Fixed pose
81  {
82  tool_pose_sampler = [](const Eigen::Isometry3d& tool_pose) {
83  return tesseract_common::VectorIsometry3d({ tool_pose });
84  };
85  }
86  else if ((wp->getCoefficients().head(3).array() > 0).all() && !(wp->getCoefficients()(3) > 0) &&
87  (wp->getCoefficients()(4) > 0) && (wp->getCoefficients()(5) > 0))
88  {
89  tool_pose_sampler =
90  std::bind(&tesseract_motion_planners::sampleToolXAxis, std::placeholders::_1, x_axes_sample_resolution);
91  }
92  else if ((wp->getCoefficients().head(3).array() > 0).all() && (wp->getCoefficients()(3) > 0) &&
93  !(wp->getCoefficients()(4) > 0) && (wp->getCoefficients()(5) > 0))
94  {
95  tool_pose_sampler =
96  std::bind(&tesseract_motion_planners::sampleToolYAxis, std::placeholders::_1, y_axes_sample_resolution);
97  }
98  else if ((wp->getCoefficients().head(3).array() > 0).all() && (wp->getCoefficients()(3) > 0) &&
99  (wp->getCoefficients()(4) > 0) && !(wp->getCoefficients()(5) > 0))
100  {
101  tool_pose_sampler =
102  std::bind(&tesseract_motion_planners::sampleToolZAxis, std::placeholders::_1, z_axes_sample_resolution);
103  }
104  else
105  {
106  CONSOLE_BRIDGE_logError("Tesseract Descartes planner does not support the provided under constrained cartesian "
107  "pose!");
108  }
109  }
110 
111  return tool_pose_sampler;
112 }
