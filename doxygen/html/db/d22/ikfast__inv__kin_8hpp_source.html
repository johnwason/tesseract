53 bool IKFastInvKin::calcInvKin(Eigen::VectorXd& solutions,
54  const Eigen::Isometry3d& pose,
55  const Eigen::Ref<const Eigen::VectorXd>& seed) const
56 {
57  // Convert to ikfast data type
58  Eigen::Transform<IkReal, 3, Eigen::Isometry> ikfast_tcp = pose.cast<IkReal>();
59 
60  // Decompose
61  const Eigen::Matrix<IkReal, 3, 1> translation = ikfast_tcp.translation();
62 
63  // Note the row major ordering here: IkFast expects the matrix in r00, r01, r02, ..., r11, r12, r13
64  // ordering
65  const Eigen::Matrix<IkReal, 3, 3, Eigen::RowMajor> rotation = ikfast_tcp.rotation();
66 
67  // Call IK (TODO: Make a better solution list class? One that uses vector instead of list)
68  ikfast::IkSolutionList<IkReal> ikfast_solution_set;
69  ComputeIk(translation.data(), rotation.data(), nullptr, ikfast_solution_set);
70 
71  // Unpack the solutions into the output vector
72  const auto n_sols = ikfast_solution_set.GetNumSolutions();
73  int ikfast_dof = numJoints();
74 
75  std::vector<IkReal> ikfast_output;
76  ikfast_output.resize(n_sols * ikfast_dof);
77 
78  for (std::size_t i = 0; i < n_sols; ++i)
79  {
80  // This actually walks the list EVERY time from the start of i.
81  const auto& sol = ikfast_solution_set.GetSolution(i);
82  auto* out = ikfast_output.data() + i * ikfast_dof;
83  sol.GetSolution(out, nullptr);
84  }
85 
86  std::vector<double> sols;
87  sols.insert(end(sols), std::make_move_iterator(ikfast_output.begin()), std::make_move_iterator(ikfast_output.end()));
88 
89  // Check the output
90  int num_sol = sols.size() / ikfast_dof;
91  std::vector<double> solution_set;
92  solution_set.reserve(sols.size());
93  for (int i = 0; i < num_sol; i++)
94  {
95  double* sol = sols.data() + ikfast_dof * i;
96  if (isValid<double>(sol, ikfast_dof))
97  {
98  harmonizeTowardZero<double>(sol, ikfast_dof); // Modifies 'sol' in place
99 
100  // Add solution
101  if (isWithinLimits<double>(Eigen::Map<Eigen::VectorXd>(sol, ikfast_dof), joint_limits_))
102  solution_set.insert(end(solution_set), sol, sol + ikfast_dof);
103 
104  // Add redundant solutions
105  std::vector<double> redundant_sols = getRedundantSolutions(sol, joint_limits_);
106  if (!redundant_sols.empty())
107  {
108  int num_sol = redundant_sols.size() / ikfast_dof;
109  for (int s = 0; s < num_sol; ++s)
110  {
111  double* redundant_sol = redundant_sols.data() + ikfast_dof * s;
112  solution_set.insert(end(solution_set),
113  std::make_move_iterator(redundant_sol),
114  std::make_move_iterator(redundant_sol + ikfast_dof));
115  }
116  }
117  }
118  }
119 
120  solutions = Eigen::Map<Eigen::VectorXd>(solution_set.data(), solution_set.size());
121  return !solution_set.empty();
122 }
