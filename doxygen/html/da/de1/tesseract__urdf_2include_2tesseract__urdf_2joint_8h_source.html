121 inline tesseract_common::StatusCode::Ptr parse(tesseract_scene_graph::Joint::Ptr& joint,
122  const tinyxml2::XMLElement* xml_element,
123  const int version)
124 {
125  joint = nullptr;
126  auto status_cat = std::make_shared<JointStatusCategory>();
127  using SC = tesseract_common::StatusCode;
128 
129  // get joint name
130  std::string joint_name;
131  if (tesseract_common::QueryStringAttribute(xml_element, "name", joint_name) != tinyxml2::XML_SUCCESS)
132  return std::make_shared<SC>(JointStatusCategory::ErrorMissingAttributeName, status_cat);
133 
134  status_cat = std::make_shared<JointStatusCategory>(joint_name);
135 
136  // create joint
137  auto j = std::make_shared<tesseract_scene_graph::Joint>(joint_name);
138 
139  // get joint origin
140  const tinyxml2::XMLElement* origin = xml_element->FirstChildElement("origin");
141  if (origin != nullptr)
142  {
143  auto status = parse(j->parent_to_joint_origin_transform, origin, version);
144  if (!(*status))
145  return std::make_shared<SC>(JointStatusCategory::ErrorParsingOrigin, status_cat, status);
146  }
147 
148  // get parent link
149  const tinyxml2::XMLElement* parent = xml_element->FirstChildElement("parent");
150  if (parent == nullptr)
151  return std::make_shared<SC>(JointStatusCategory::ErrorMissingParentElement, status_cat);
152 
153  if (tesseract_common::QueryStringAttribute(parent, "link", j->parent_link_name) != tinyxml2::XML_SUCCESS)
154  return std::make_shared<SC>(JointStatusCategory::ErrorParsingParentAttributeLink, status_cat);
155 
156  // get child link
157  const tinyxml2::XMLElement* child = xml_element->FirstChildElement("child");
158  if (child == nullptr)
159  return std::make_shared<SC>(JointStatusCategory::ErrorMissingChildElement, status_cat);
160 
161  if (tesseract_common::QueryStringAttribute(child, "link", j->child_link_name) != tinyxml2::XML_SUCCESS)
162  return std::make_shared<SC>(JointStatusCategory::ErrorParsingChildAttributeLink, status_cat);
163 
164  // get joint type
165  std::string joint_type;
166  if (tesseract_common::QueryStringAttribute(xml_element, "type", joint_type) != tinyxml2::XML_SUCCESS)
167  return std::make_shared<SC>(JointStatusCategory::ErrorMissingTypeElement, status_cat);
168 
169  if (joint_type == "planar")
170  j->type = tesseract_scene_graph::JointType::PLANAR;
171  else if (joint_type == "floating")
172  j->type = tesseract_scene_graph::JointType::FLOATING;
173  else if (joint_type == "revolute")
174  j->type = tesseract_scene_graph::JointType::REVOLUTE;
175  else if (joint_type == "continuous")
176  j->type = tesseract_scene_graph::JointType::CONTINUOUS;
177  else if (joint_type == "prismatic")
178  j->type = tesseract_scene_graph::JointType::PRISMATIC;
179  else if (joint_type == "fixed")
180  j->type = tesseract_scene_graph::JointType::FIXED;
181  else
182  return std::make_shared<SC>(JointStatusCategory::ErrorInvalidType, status_cat);
183 
184  // get joint axis
185  if (j->type != tesseract_scene_graph::JointType::FLOATING && j->type != tesseract_scene_graph::JointType::FIXED)
186  {
187  const tinyxml2::XMLElement* axis = xml_element->FirstChildElement("axis");
188  if (axis == nullptr)
189  {
190  j->axis = Eigen::Vector3d(1.0, 0.0, 0.0);
191  }
192  else
193  {
194  std::string axis_str;
195  if (tesseract_common::QueryStringAttribute(axis, "xyz", axis_str) != tinyxml2::XML_SUCCESS)
196  return std::make_shared<SC>(JointStatusCategory::ErrorParsingAxisAttributeXYZ, status_cat);
197 
198  std::vector<std::string> tokens;
199  boost::split(tokens, axis_str, boost::is_any_of(" "), boost::token_compress_on);
200  if (tokens.size() != 3 || !tesseract_common::isNumeric(tokens))
201  return std::make_shared<SC>(JointStatusCategory::ErrorParsingAxisAttributeXYZ, status_cat);
202 
203  double ax, ay, az;
204  // No need to check return values because the tokens are verified above
205  tesseract_common::toNumeric<double>(tokens[0], ax);
206  tesseract_common::toNumeric<double>(tokens[1], ay);
207  tesseract_common::toNumeric<double>(tokens[2], az);
208 
209  j->axis = Eigen::Vector3d(ax, ay, az);
210  }
211  }
212 
213  // get joint limits
214  if (j->type == tesseract_scene_graph::JointType::REVOLUTE || j->type == tesseract_scene_graph::JointType::PRISMATIC ||
215  j->type == tesseract_scene_graph::JointType::CONTINUOUS)
216  {
217  const tinyxml2::XMLElement* limits = xml_element->FirstChildElement("limit");
218  if (limits == nullptr && j->type != tesseract_scene_graph::JointType::CONTINUOUS)
219  {
220  return std::make_shared<SC>(JointStatusCategory::ErrorMissingLimitsElement, status_cat);
221  }
222 
223  if (limits == nullptr && j->type == tesseract_scene_graph::JointType::CONTINUOUS)
224  {
225  j->limits = std::make_shared<tesseract_scene_graph::JointLimits>();
226  }
227  else
228  {
229  auto status = parse(j->limits, limits, version);
230  if (!(*status))
231  return std::make_shared<SC>(JointStatusCategory::ErrorParsingLimits, status_cat, status);
232  }
233  }
234 
235  // get joint safety if exists
236  const tinyxml2::XMLElement* safety = xml_element->FirstChildElement("safety_controller");
237  if (safety != nullptr)
238  {
239  auto status = parse(j->safety, safety, version);
240  if (!(*status))
241  return std::make_shared<SC>(JointStatusCategory::ErrorParsingSafetyController, status_cat, status);
242  }
243 
244  // get joint calibration if exists
245  const tinyxml2::XMLElement* calibration = xml_element->FirstChildElement("calibration");
246  if (calibration != nullptr)
247  {
248  auto status = parse(j->calibration, calibration, version);
249  if (!(*status))
250  return std::make_shared<SC>(JointStatusCategory::ErrorParsingCalibration, status_cat, status);
251  }
252 
253  // get mimic joint if exists
254  const tinyxml2::XMLElement* mimic = xml_element->FirstChildElement("mimic");
255  if (mimic != nullptr)
256  {
257  auto status = parse(j->mimic, mimic, version);
258  if (!(*status))
259  return std::make_shared<SC>(JointStatusCategory::ErrorParsingMimic, status_cat, status);
260  }
261 
262  // get dynamics if exists
263  const tinyxml2::XMLElement* dynamics = xml_element->FirstChildElement("dynamics");
264  if (dynamics != nullptr)
265  {
266  auto status = parse(j->dynamics, dynamics, version);
267  if (!(*status))
268  return std::make_shared<SC>(JointStatusCategory::ErrorParsingDynamics, status_cat, status);
269  }
270 
271  joint = std::move(j);
272  return std::make_shared<SC>(JointStatusCategory::Success, status_cat);
273 }
