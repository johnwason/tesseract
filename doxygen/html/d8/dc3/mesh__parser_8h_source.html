65 template <class T>
66 inline std::vector<std::shared_ptr<T>> extractMeshData(const aiScene* scene,
67  const aiNode* node,
68  const aiMatrix4x4& parent_transform,
69  const Eigen::Vector3d& scale,
70  tesseract_common::Resource::Ptr resource)
71 {
72  std::vector<std::shared_ptr<T>> meshes;
73 
74  aiMatrix4x4 transform = parent_transform;
75  transform *= node->mTransformation;
76  for (unsigned int j = 0; j < node->mNumMeshes; ++j)
77  {
78  auto vertices = std::make_shared<tesseract_common::VectorVector3d>();
79  auto triangles = std::make_shared<Eigen::VectorXi>();
80 
81  const aiMesh* a = scene->mMeshes[node->mMeshes[j]];
82  for (unsigned int i = 0; i < a->mNumVertices; ++i)
83  {
84  aiVector3D v = transform * a->mVertices[i];
85  vertices->push_back(Eigen::Vector3d(static_cast<double>(v.x) * scale(0),
86  static_cast<double>(v.y) * scale(1),
87  static_cast<double>(v.z) * scale(2)));
88  }
89 
90  long triangle_count = 0;
91  std::vector<int> local_triangles;
92  local_triangles.reserve(a->mNumFaces);
93  for (unsigned int i = 0; i < a->mNumFaces; ++i)
94  {
95  if (a->mFaces[i].mNumIndices >= 3)
96  {
97  triangle_count += 1;
98  local_triangles.push_back(static_cast<int>(a->mFaces[i].mNumIndices));
99  for (size_t k = 0; k < a->mFaces[i].mNumIndices; ++k)
100  local_triangles.push_back(static_cast<int>(a->mFaces[i].mIndices[k]));
101  }
102  else
103  {
104  CONSOLE_BRIDGE_logDebug("Mesh had a face with less than three verticies: %s", resource->getUrl().c_str());
105  }
106  }
107 
108  triangles->resize(static_cast<long>(local_triangles.size()));
109  for (long i = 0; i < triangles->size(); ++i)
110  (*triangles)[i] = local_triangles[static_cast<size_t>(i)];
111 
112  meshes.push_back(std::make_shared<T>(vertices, triangles, static_cast<int>(triangle_count), resource, scale));
113  }
114 
115  for (unsigned int n = 0; n < node->mNumChildren; ++n)
116  {
117  std::vector<std::shared_ptr<T>> child_meshes =
118  extractMeshData<T>(scene, node->mChildren[n], transform, scale, resource);
119  meshes.insert(meshes.end(), child_meshes.begin(), child_meshes.end());
120  }
121  return meshes;
122 }

131 template <class T>
132 inline std::vector<std::shared_ptr<T>> createMeshFromAsset(const aiScene* scene,
133  const Eigen::Vector3d& scale,
134  tesseract_common::Resource::Ptr resource)
135 {
136  if (!scene->HasMeshes())
137  {
138  CONSOLE_BRIDGE_logWarn("Assimp reports scene in %s has no meshes", resource->getUrl().c_str());
139  return std::vector<std::shared_ptr<T>>();
140  }
141  std::vector<std::shared_ptr<T>> meshes = extractMeshData<T>(scene, scene->mRootNode, aiMatrix4x4(), scale, resource);
142  if (meshes.empty())
143  {
144  CONSOLE_BRIDGE_logWarn("There are no meshes in the scene %s", resource->getUrl().c_str());
145  return std::vector<std::shared_ptr<T>>();
146  }
147 
148  return meshes;
149 }

161 template <class T>
162 inline std::vector<std::shared_ptr<T>> createMeshFromPath(const std::string& path,
163  Eigen::Vector3d scale = Eigen::Vector3d(1, 1, 1),
164  bool triangulate = false,
165  bool flatten = false)
166 {
167  // Create an instance of the Importer class
168  Assimp::Importer importer;
169 
170  // Issue #38 fix: as part of the post-processing, we remove all other components in file but
171  // the meshes, as anyway the resulting shapes:Mesh object just receives vertices and triangles.
172  importer.SetPropertyInteger(AI_CONFIG_PP_RVC_FLAGS,
173  aiComponent_NORMALS | aiComponent_TANGENTS_AND_BITANGENTS | aiComponent_COLORS |
174  aiComponent_TEXCOORDS | aiComponent_BONEWEIGHTS | aiComponent_ANIMATIONS |
175  aiComponent_TEXTURES | aiComponent_LIGHTS | aiComponent_CAMERAS |
176  aiComponent_MATERIALS);
177 
178  // And have it read the given file with some post-processing
179  const aiScene* scene = nullptr;
180  if (triangulate)
181  scene = importer.ReadFile(path.c_str(),
182  aiProcess_Triangulate | aiProcess_JoinIdenticalVertices | aiProcess_SortByPType |
183  aiProcess_RemoveComponent);
184  else
185  scene = importer.ReadFile(path.c_str(),
186  aiProcess_JoinIdenticalVertices | aiProcess_SortByPType | aiProcess_RemoveComponent);
187 
188  if (!scene)
189  {
190  CONSOLE_BRIDGE_logError("Could not load mesh from \"%s\": %s", path.c_str(), importer.GetErrorString());
191  return std::vector<std::shared_ptr<T>>();
192  }
193 
194  // Assimp enforces Y_UP convention by rotating models with different conventions.
195  // However, that behaviour is confusing and doesn't match the ROS convention
196  // where the Z axis is pointing up.
197  // Hopefully this doesn't undo legit use of the root node transformation...
198  // Note that this is also what RViz does internally.
199  scene->mRootNode->mTransformation = aiMatrix4x4();
200 
201  if (flatten)
202  {
203  // These post processing steps flatten the root node transformation into child nodes,
204  // so they must be delayed until after clearing the root node transform above.
205  importer.ApplyPostProcessing(aiProcess_OptimizeMeshes | aiProcess_OptimizeGraph);
206  }
207  else
208  {
209  importer.ApplyPostProcessing(aiProcess_OptimizeGraph);
210  }
211 
212  return createMeshFromAsset<T>(scene, scale, nullptr);
213 }

226 inline std::vector<std::shared_ptr<T>> createMeshFromResource(tesseract_common::Resource::Ptr resource,
227  Eigen::Vector3d scale = Eigen::Vector3d(1, 1, 1),
228  bool triangulate = false,
229  bool flatten = false)
230 {
231  if (!resource)
232  return std::vector<std::shared_ptr<T>>();
233 
234  const char* hint = nullptr;
235 
236  std::string resource_url = resource->getUrl();
237  std::regex hint_re("^.*\\.([A-Za-z0-9]{1,8})$");
238  std::smatch hint_match;
239  if (std::regex_match(resource_url, hint_match, hint_re))
240  {
241  if (hint_match.size() == 2)
242  {
243  hint = hint_match[1].str().c_str();
244  }
245  }
246 
247  std::vector<uint8_t> data = resource->getResourceContents();
248  if (data.empty())
249  {
250  if (resource->isFile())
251  return createMeshFromPath<T>(resource->getFilePath(), scale, triangulate, flatten);
252 
253  return std::vector<std::shared_ptr<T>>();
254  }
255 
256  // Create an instance of the Importer class
257  Assimp::Importer importer;
258 
259  // Issue #38 fix: as part of the post-processing, we remove all other components in file but
260  // the meshes, as anyway the resulting shapes:Mesh object just receives vertices and triangles.
261  importer.SetPropertyInteger(AI_CONFIG_PP_RVC_FLAGS,
262  aiComponent_NORMALS | aiComponent_TANGENTS_AND_BITANGENTS | aiComponent_COLORS |
263  aiComponent_TEXCOORDS | aiComponent_BONEWEIGHTS | aiComponent_ANIMATIONS |
264  aiComponent_TEXTURES | aiComponent_LIGHTS | aiComponent_CAMERAS |
265  aiComponent_MATERIALS);
266 
267  // And have it read the given file with some post-processing
268  const aiScene* scene = nullptr;
269  if (triangulate)
270  scene = importer.ReadFileFromMemory(&data[0],
271  data.size(),
272  aiProcess_Triangulate | aiProcess_JoinIdenticalVertices |
273  aiProcess_SortByPType | aiProcess_RemoveComponent,
274  hint);
275  else
276  scene =
277  importer.ReadFileFromMemory(&data[0],
278  data.size(),
279  aiProcess_JoinIdenticalVertices | aiProcess_SortByPType | aiProcess_RemoveComponent,
280  hint);
281 
282  if (!scene)
283  {
284  CONSOLE_BRIDGE_logError(
285  "Could not load mesh from \"%s\": %s", resource->getUrl().c_str(), importer.GetErrorString());
286  return std::vector<std::shared_ptr<T>>();
287  }
288

290  // However, that behaviour is confusing and doesn't match the ROS convention
291  // where the Z axis is pointing up.
292  // Hopefully this doesn't undo legit use of the root node transformation...
293  // Note that this is also what RViz does internally.
294  scene->mRootNode->mTransformation = aiMatrix4x4();

297  {
298  // These post processing steps flatten the root node transformation into child nodes,
299  // so they must be delayed until after clearing the root node transform above.
300  importer.ApplyPostProcessing(aiProcess_OptimizeMeshes | aiProcess_OptimizeGraph);
301  }
302  else
303  {
304  importer.ApplyPostProcessing(aiProcess_OptimizeGraph);
305  }
306 
307  return createMeshFromAsset<T>(scene, scale, resource);
