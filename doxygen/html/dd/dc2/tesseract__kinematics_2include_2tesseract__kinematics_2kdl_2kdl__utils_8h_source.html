162 inline bool parseSceneGraph(KDLChainData& results,
163  const tesseract_scene_graph::SceneGraph& scene_graph,
164  const std::vector<std::pair<std::string, std::string>>& chains)
165 {
166  if (!tesseract_scene_graph::parseSceneGraph(scene_graph, results.kdl_tree))
167  {
168  CONSOLE_BRIDGE_logError("Failed to parse KDL tree from Scene Graph");
169  return false;
170  }
171 
172  results.chains = chains;
173  results.base_name = chains.front().first;
174  for (const auto& chain : chains)
175  {
176  KDL::Chain sub_chain;
177  if (!results.kdl_tree.getChain(chain.first, chain.second, sub_chain))
178  {
179  CONSOLE_BRIDGE_logError(
180  "Failed to initialize KDL between links: '%s' and '%s'", chain.first.c_str(), chain.second.c_str());
181  return false;
182  }
183  results.robot_chain.addChain(sub_chain);
184  }
185  results.tip_name = chains.back().second;
186 
187  results.joint_list.resize(results.robot_chain.getNrOfJoints());
188  results.joint_limits.resize(results.robot_chain.getNrOfJoints(), 2);
189 
190  results.segment_index[results.base_name] = 0;
191  results.link_list.push_back(results.base_name);
192  results.active_link_list.clear();
193  bool found = false;
194  for (unsigned i = 0, j = 0; i < results.robot_chain.getNrOfSegments(); ++i)
195  {
196  const KDL::Segment& seg = results.robot_chain.getSegment(i);
197  const KDL::Joint& jnt = seg.getJoint();
198  results.link_list.push_back(seg.getName());
199 
200  // KDL segments does not contain the the base link in this list. When calling function that take segmentNr, like
201  // JntToCart to get the base link transform you would pass an index of zero and for subsequent links it is
202  // index + 1. This was determined through testing which is captured in this packages unit tests.
203  results.segment_index[seg.getName()] = static_cast<int>(i + 1);
204 
205  if (found)
206  results.active_link_list.push_back(seg.getName());
207 
208  if (jnt.getType() == KDL::Joint::None)
209  continue;
210 
211  if (!found)
212  {
213  found = true;
214  results.active_link_list.push_back(seg.getName());
215  }
216 
217  results.joint_list[j] = jnt.getName();
218  const tesseract_scene_graph::Joint::ConstPtr& joint = scene_graph.getJoint(jnt.getName());
219  results.joint_limits(j, 0) = joint->limits->lower;
220  results.joint_limits(j, 1) = joint->limits->upper;
221 
222  // Need to set limits for continuous joints. TODO: This may not be required
223  // by the optimization library but may be nice to have
224  if (joint->type == tesseract_scene_graph::JointType::CONTINUOUS &&
225  std::abs(results.joint_limits(j, 0) - results.joint_limits(j, 1)) <=
226  static_cast<double>(std::numeric_limits<float>::epsilon()))
227  {
228  results.joint_limits(j, 0) = -4 * M_PI;
229  results.joint_limits(j, 1) = +4 * M_PI;
230  }
231  ++j;
232  }
233 
234  return true;
235 }
